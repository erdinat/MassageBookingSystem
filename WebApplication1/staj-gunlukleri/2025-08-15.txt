TEKNİK DETAYLAR
==========================================

1. Stored Procedure:
```sql
CREATE PROCEDURE GetTherapistReport
    @TherapistId INT = NULL,
    @StartDate DATETIME = NULL
AS
BEGIN
    SELECT t.Name AS TherapistName,
           c.Name + ' ' + c.Surname AS Customer,
           s.Name AS Service,
           av.StartTime,
           av.EndTime,
           a.CreatedAt AS AppointmentCreated
    FROM Appointments a
    JOIN Therapists t ON a.TherapistId = t.Id
    JOIN Customers c ON a.CustomerId = c.Id
    JOIN Services s ON a.ServiceId = s.Id
    JOIN AvailabilitySlots av ON a.AvailabilitySlotId = av.Id
    WHERE (@TherapistId IS NULL OR t.Id = @TherapistId)
      AND (@StartDate IS NULL OR av.StartTime >= @StartDate)
    ORDER BY av.StartTime DESC;
END;
```

2. View Definition:
```sql
CREATE VIEW AppointmentSummary AS 
SELECT 
    a.Id,
    c.Name + ' ' + c.Surname AS Customer,
    t.Name AS Therapist,
    s.Name AS Service,
    av.StartTime,
    av.EndTime,
    CASE 
        WHEN av.StartTime > GETUTCDATE() THEN 'Gelecek'
        ELSE 'Geçmiş'
    END AS Status
FROM Appointments a
JOIN Customers c ON a.CustomerId = c.Id
JOIN Therapists t ON a.TherapistId = t.Id
JOIN Services s ON a.ServiceId = s.Id
JOIN AvailabilitySlots av ON a.AvailabilitySlotId = av.Id;
```
